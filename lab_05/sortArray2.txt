#include <iostream>#include <fstream>#include <string>using namespace std;int binarySearchL(string* array, int size, string value) {	int first = 0;    int last = size-1;    while (first <= last) {        int middle = (first + last)/2;   if middle is less than value            //look to the right            first = middle+1;            middle = (first + last) / 2;        } else if middle is greater than value            //look to the left            last = middle-1;            middle = (first + last) / 2;        } else if middle equals value {            return middle;        }    }    return -1;}
